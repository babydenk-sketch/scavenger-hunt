        <p id="clueText">Welcome! Tap "Start" to get your first clue.</p>

        <button onclick="startHunt()">Start Hunt</button>

        <div id="nextBtn" style="display:none;">

            <button onclick="navigateToNext()">Open Google Maps Directions</button>

        </div>

    <p id="status"></p>


    <script>

        // === CUSTOMIZE HERE ===

        const locations = [

            {

                id: 1,

                lat: 40.7128,      // Example: Statue of Liberty area

                lng: -74.0060,

                clue: "Find the famous green lady standing tall in the harbor. What is her full official name?",

                question: "You've arrived! Answer: What year was she dedicated?",

                answerHint: "1886"  // Optional - you can check answer manually or remove

            },

            {

                id: 2,

                lat: 40.7484,

                lng: -73.9857,     // Empire State Building

                clue: "Head to the building that was once the tallest in the world. Look for the iconic spire!",

                question: "Great job! Now tell me: How many floors does this building have?"

            },

            {

                id: 3,

                lat: 40.7580,

                lng: -73.9855,     // Times Square

                clue: "Go to the place where the bright lights never sleep. Find the red steps!",

                question: "You're there! What is the name of the famous square you're standing in?"

            },

            // Add more locations here (copy-paste the format)

        ];


        let currentStage = 0;

        let watchId = null;

        const proximityRadius = 50; // meters


        function haversineDistance(lat1, lon1, lat2, lon2) {

            const R = 6371000; // Earth radius in meters

            const dLat = (lat2 - lat1) * Math.PI / 180;

            const dLon = (lon2 - lon1) * Math.PI / 180;

            const a = Math.sin(dLat/2) * Math.sin(dLat/2) +

                      Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *

                      Math.sin(dLon/2) * Math.sin(dLon/2);

            const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));

            return R * c;

        }


        function checkLocation(position) {

            const playerLat = position.coords.latitude;

            const playerLng = position.coords.longitude;

            const target = locations[currentStage];


            const distance = haversineDistance(playerLat, playerLng, target.lat, target.lng);

            const statusEl = document.getElementById("status");


            if (distance <= proximityRadius) {

                statusEl.innerHTML = `✅ You're at the spot! New question unlocked.`;

                showNextQuestion();

            } else {

                statusEl.innerHTML = `📍 ${Math.round(distance)} meters away from next spot. Keep going!`;

            }

        }


        function showNextQuestion() {

            currentStage++;

            const target = locations[currentStage - 1]; // just arrived at this one

            document.getElementById("clueTitle").textContent = `Question ${currentStage}`;

            document.getElementById("clueText").innerHTML = `

                <strong>You made it!</strong><br>

                ${target.question || "Next clue coming..."}<br><br>

                <em>Clue for next spot:</em> ${locations[currentStage] ? locations[currentStage].clue : "<strong>🎉 You finished the hunt!</strong>"}

            `;

            document.getElementById("nextBtn").style.display = locations[currentStage] ? "block" : "none";


            if (!locations[currentStage]) {

                // Final screen

                document.getElementById("clueText").innerHTML += `<br><br>🏆 Congratulations! You completed the scavenger hunt!`;

            }

        }


        function navigateToNext() {

            if (currentStage >= locations.length) return;

            const next = locations[currentStage];

            const url = `https://www.google.com/maps/dir/?api=1&destination=${next.lat},${next.lng}&travelmode=walking`;

            window.open(url, "_blank");

        }


        function startHunt() {

            if (currentStage === 0) {

                currentStage = 0;

                document.getElementById("clueTitle").textContent = "First Clue";

                document.getElementById("clueText").innerHTML = `

                    ${locations[0].clue}<br><br>

                    <em>Tap the button below once you're close to get directions.</em>

                `;

                document.getElementById("nextBtn").style.display = "block";


                // Start watching GPS

                if (navigator.geolocation) {

                    watchId = navigator.geolocation.watchPosition(checkLocation, 

                        () => document.getElementById("status").textContent = "⚠️ Enable location in your browser settings",

                        { enableHighAccuracy: true, maximumAge: 10000 }

                    );

                } else {

                    alert("Your browser doesn't support GPS location.");

                }

            }

        }


        // Optional: stop GPS when page closes

        window.onbeforeunload = () => {

            if (watchId) navigator.geolocation.clearWatch(watchId);

        };

    </script>
